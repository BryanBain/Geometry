\textsc{Parallelograms} 

\begin{itemize}
    \item Opposite sides are parallel
    \item Opposite sides are congruent
    \item Opposite angles are congruent
    \item Consecutive angles are supplementary
    \item Diagonals bisect each other
\end{itemize}
