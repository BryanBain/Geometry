    \tkzLabelPoints[left](A)
    \tkzLabelPoints[right](C)
    \tkzLabelPoints[above](B)
    \tkzDefLine[orthogonal = through B](A,C)
    \tkzGetPoint{D}
    \tkzInterLL(B,D)(A,C)
    \tkzGetPoint{E}
    \tkzDrawSegment[dashed](B,E)
    \tkzMarkRightAngle(C,E,B)
